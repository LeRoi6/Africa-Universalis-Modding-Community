		random_owned_province = {
			limit = {
				has_estate = $ESTATE$
			}
			save_event_target_as = $EVENTTARGET$
		}
	}
	if = {
		limit = {
			monthly_income = 25
			has_saved_event_target = $EVENTTARGET$
			estate_influence = {
				estate = $ESTATE$
				influence = 80
			}
		}
		define_advisor = {
			type = $ADVISORTYPE$
