#Hussites march on Prague
country_event = {
	id = flavor_boh.1
	title = flavor_boh.1.t
	desc = flavor_boh.1.d
	picture = CIVIL_WAR_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = BOH
			AND = {
				tag = GMV
				primary_culture = czech
				NOT = { exists = BOH }
			}
		}
		owns = 266
		is_year = 1446
		NOT = { is_year = 1457 }
		is_at_war = no
		religion = catholic
		is_free_or_tributary_trigger = yes
	}
	
	mean_time_to_happen = {
		months = 17
	}
	
	option = {
		ai_chance = { factor = 90 }
		name = flavor_boh.1.a
		add_papal_influence = -100
		if = {
			limit = {
				has_regency = no
			}
			tooltip = {
				kill_ruler = yes
			}
		}
		if = {
			limit = {
				has_heir = yes
			}
			remove_heir = yes
		}
		define_ruler = {
			name = "Jirí"
			dynasty = "z Podebrad"
			adm = 5
			dip = 3
			mil = 3
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = zealot_personality }
		}
		add_reform_desire = 0.03
		add_ruler_modifier = {
			name = "boh_hussite_king"
			duration = -1
		}
		set_country_flag = boh_supported_hussites
		set_country_flag = boh_hussites_mission_accept
		PAP = {
			country_event = { id = flavor_boh.9 }
		}
		every_known_country = {
			limit = {
				religion = catholic
				is_free_or_tributary_trigger = yes
				OR = {
					is_neighbor_of = ROOT
					num_of_cities = 15
					government = theocracy
				}
			}
			add_opinion = {
				who = root
				modifier = boh_heretic_hussite
			}
		}
	}
	
	option = {
		name = flavor_boh.1.b
		266 = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				leader = "Jirí z Podebrad"
			}		
		}
		random_owned_province = {
			limit = {
				NOT = { province_id = 266 }
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 1
			}
		}
		set_country_flag = boh_hussites_mission_decline
		if = {
			limit = { exists = PAP }
			add_papal_influence = 5
			PAP = {
				add_opinion = {
					who = root
					modifier = boh_papal_reconciled
				}
			}
		}
	}
}

#The Heretic King
country_event = {
	id = flavor_boh.2
	title = flavor_boh.2.t
	desc = flavor_boh.2.d
	picture = COURT_eventPicture
	
	is_triggered_only = yes

	trigger = {
		OR = {
			tag = BOH
			AND = {
				tag = GMV
				primary_culture = czech
				NOT = { exists = BOH }
			}
		}
		has_country_flag = boh_supported_hussites
		NOT = { is_religion_enabled = protestant }
		has_regency = no
	}
	
	immediate = {
		every_country = {
			limit = {
				has_opinion_modifier = {
					who = ROOT
					modifier = boh_heretic_hussite
				}
			}
			remove_opinion = {
				who = ROOT
				modifier = boh_heretic_hussite
			}
		}
	}
	
	option = {
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0.1
				is_year = 1470
			}
		}
		name = flavor_boh.2.a
		add_stability = -1
		if = {
			limit = { exists = PAP }
			add_papal_influence = 5
			PAP = {
				add_opinion = {
					who = root
					modifier = boh_papal_reconciled
				}
			}
		}
		add_reform_desire = -0.01
		clr_country_flag = boh_supported_hussites
		if = {
			limit = {
				NOT = { is_year = 1490 }
			}
			random_owned_province = {
				limit = { is_state = yes }
				spawn_rebels = {
					type = heretic_rebels
					size = 2
				}
			}
		}
	}
	option = {
		name = flavor_boh.2.b
		ai_chance = {
			factor = 10
		}
		add_ruler_modifier = {
			name = "boh_hussite_king"
			duration = -1
		}
		if = {
			limit = {
				exists = PAP
			}
			PAP = { country_event = { id = flavor_boh.9 } }
		}
		add_reform_desire = 0.01
		custom_tooltip = flavor_boh.2.3
		every_known_country = {
			limit = {
				religion = catholic
				is_free_or_tributary_trigger = yes
				OR = {
					is_neighbor_of = ROOT
					num_of_cities = 15
					government = theocracy
				}
			}
			add_opinion = {
				who = root
				modifier = boh_heretic_hussite
			}
		}
	}
}

# Winter King in Prague
country_event = {
	id = flavor_boh.8
	title = flavor_boh.8.t
	desc = flavor_boh.8.d
	picture = COURT_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			tag = BOH
			AND = {
				tag = GMV
				primary_culture = czech
				NOT = { exists = BOH }
			}
		}
		is_emperor = no
		religion = catholic
		any_owned_province = {
			religion = protestant
			culture = czech
		}
		is_year = 1580
	}
	
	
	option = {
		name = flavor_boh.8.a
		ai_chance = { factor = 90 }	
		change_religion = protestant
		if = {
			limit = {
				emperor = { religion = catholic }
			}
			emperor = {
				add_casus_belli = {	
					target = BOH
					type = cb_religious_conformance
					months = 1200
				}
				add_opinion = {
					who = root
					modifier = boh_protestant_king
				}										
			}
		}

		every_neighbor_country = {
			limit = {
				is_emperor = no
				religion = catholic
				num_of_cities = 5
			}
			add_casus_belli = {	
				target = BOH
				type = cb_religious_conformance
				months = 1200
			}	
			add_opinion = {
				who = root
				modifier = boh_protestant_king
			}							
		}
	}
}
